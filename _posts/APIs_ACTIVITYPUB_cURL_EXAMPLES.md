```bash

--header 'Authorization: Bearer <TOKEN>' \
--header 'Content-Type: application/json' \
--data-raw '{
    "name": "HVTURINGGA",
    "avatar": "https://avatars.githubusercontent.com/u/35920389?v=4",
    "summary": "bio"
}'
```

}
